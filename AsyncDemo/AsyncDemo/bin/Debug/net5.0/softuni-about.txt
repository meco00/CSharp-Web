        var duration = 500;
        $(window).scroll(function () {
            if ($(this).scrollTop() > offset) {
                $('#scrollToTop').fadeIn(duration);
            } else {
                $('#scrollToTop').fadeOut(duration);
            }
        });

        $('#scrollToTop').click(function (event) {
            event.preventDefault();
            $('html, body').animate({ scrollTop: 0 }, 800);
            return false;
        });
    });
</script>
